\section*{Eratosthenes Sieve}
\begin{flushleft}
 \begin{verbatim}
  isPrime :: [Integer] Integer -> Bool
  isPrime lp n 
           | (head lp) ^ 2 > n = True
           | mod n (head lp) == 0 False
           | otherwise = isPrime(tail lp) n
 
  
  primes = 2 : (filter (isPrime primes) [3,5...])
 \end{verbatim}
\end{flushleft}
